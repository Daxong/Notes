### 1.吃豆子过桥问题
[吃豆子过桥问题](https://blog.csdn.net/yjf3151731373/article/details/52180371)
```
#include <bits/stdc++.h>

using namespace std;

// f(n,m) = f(n - 1, m) + 2 * T + 1 (T表示往返次数), 其中 T = ceil( f(n-1,m)-(m-1)) / (m-2) )
// f(n-1,m) + ceil( f(n-1,m)-(m-1)) / (m-2) ) * 2+ 1
// = f(n-1,m) + ( f(n-1,m) - 2) / (m-2) ) * 2 + 1 (len < max_num)

int get_min_num(int len, int max_num) {
  if(len <= max_num){
    return len;
  }
  int ans = max_num;
  for(int i = max_num; i < len; i++) {
    ans = ans + ((ans - 2) / (max_num - 2) ) * 2 + 1;
  }
  return ans;
}
// 原点有无数豆子, 可以放豆子，可以折返，有最大可带豆子数
int main(int argc, char const *argv[]) {
  int max_num = 0;
  int len = 0;
  // len 为路的长度, max_num 为最大可带的豆子数
  std::cin >> len >> max_num;
  std::cout << get_min_num(len , max_num) << '\n'; // 最少需要吃多少颗豆子才能走完路
  return 0;
}
```
